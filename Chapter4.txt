3) Polygon

def polygon(t,length,n):
	for i in range(n):
		fd(t,length)
		lt(t,360/n)

polygon(bob,50,5)

4) Circle

def circle(t,r):
	t.delay = 0.01
	c = 2.0*3.14159*r
	n = 100
	length = c/n
	for i in range(n):
		fd(t,length)
		lt(t,360.0/n)

circle(bob,50)

5) Arc

def arc(t,r,a):
	t.delay = 0.01
	c = 2.0*3.14159*r
	arc = c*a/360
	n = int(arc/3)+1
	length = arc/n
	for i in range(n):
		fd(t,length)
		lt(t,float(a)/n)

arc(bob,50,30)
